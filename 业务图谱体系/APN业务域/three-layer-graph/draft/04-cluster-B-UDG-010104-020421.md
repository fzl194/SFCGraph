### 8.1 GWFD-010104 外部地址分配方式（来源：激活地址分配方式_71653959.md）

```
//配置外部地址分配时的白名单检测功能。
ADD VPNINST:VPNINSTANCE="vpn1";
ADD POOLGROUP: POOLGRPNAME="poolgroup1";
ADD POOL:POOLNAME="testpool",POOLTYPE=EXTERNAL,HASVPN=ENABLE,VPNINSTANCE="vpn1";
ADD SECTION:POOLNAME="testpool",SECTIONNUM=1,IPVERSION=IPV4,V4STARTIP="10.10.1.1",V4ENDIP="10.10.1.10";
ADD POOLBINDGROUP:POOLGROUPNAME="poolgroup1", POOLNAME="testpool";
ADD APN:APN="apn-test",HASVPN=ENABLE,VPNINSTANCE="vpn1";
SET APNADDRESSATTR: APN="apn-test", HOSTROUTEIP=ENABLE;
ADD POOLGRPMAP: MAPPINGNAME="mapping1", APN="apn-test", POOLGROUPNAME="poolgroup1";

//配置手机下行路由（OSPF 族）。
ADD L3VPNINST:VRFNAME="vpn1";
ADD VPNINSTAF:VRFNAME="vpn1",AFTYPE=ipv4uni;
ADD OSPF:PROCID=100,VRFNAME="vpn1",SCHEMAROUID="10.10.10.1";

```

### 8.2 GWFD-020421 基于位置分配地址（来源：基于位置分配地址_02846773.md，LAC 单位置区）

//配置地址池绑定到地址池组。
ADD POOLGROUP: POOLGRPNAME="poolgroup1";

//配置LAC位置区组与地址池组的映射关系。
