## [操作步骤](#ZH-CN_OPI_0000001980953432)

1. 配置阻断业务使用过滤规则。
    a. 配置阻断业务使用的三四层过滤规则。
      **ADD FILTER**
      **ADD FILTER** **IPV6**
    b. 配置流过滤器。
      **ADD FLOWFILTER**
    c. 绑定三四层过滤信息。
      **ADD FLTBINDFLOWF**
    d. 配置流过滤器中的协议类型。
      **ADD PROTBINDFLOWF**
2. 配置报文的阻塞动作。
  **ADD PCCACTIONPROP**
3. 配置PCC策略组。
  **ADD PCCPOLICYGRP**
4. 配置使用的规则。
  **ADD RULE**
5. **可选：**如果PCRF/PCF下发预定义规则组或没有PCRF/PCF而使用本地配置的静态规则，需要配置业务策略组合。
    a. 创建UserProfile。
      **ADD USERPROFILE**
    b. 将已配置的Rule绑定到UserProfile。
      **ADD RULEBINDING**
6. 将新配置的Filter置为生效
  **SET REFRESHSRV**

## [任务示例](#ZH-CN_OPI_0000001980953432)

任务描述

Proxy SGW-U对业务报文进行加密检测，对非加密的标准IMS业务报文正常转发、对加密业务报文进行阻断。

脚本

//安装本地User Profile。

```
SET SOFTPARA:DT=BIT,BITNUM=1559,BITVALUE=1;
```

//配置三四层过滤规则。

```
ADD FILTER:FILTERNAME="f_any", L34PROTTYPE=STRING, L34PROTOCOL=ANY, MSSTARTPORT=0, MSENDPORT=65535, SVRSTARTPORT=0, SVRENDPORT=65535
ADD FILTERIPV6:FILTERNAME="f_any_v6", L34PROTTYPEV6=STRING, L34PROTOCOL=ANY, MSSTARTPORT=0, MSENDPORT=65535, SVRSTARTPORT=0, SVRENDPORT=65535;
ADD FLOWFILTER:FLOWFILTERNAME="ff_any_ims_drop", TETHERDETFLAG=NONE, FLOWFILTERTYPE=USERCONFIG;
ADD FLOWFILTER:FLOWFILTERNAME="ff_any_ims_paas", TETHERDETFLAG=NONE, FLOWFILTERTYPE=USERCONFIG;
ADD FLTBINDFLOWF:FLOWFILTERNAME="ff_any_ims_drop", FILTERNAME="f_any";
ADD FLTBINDFLOWF:FLOWFILTERNAME="ff_any_ims_drop", FILTERNAME="f_any_v6";
ADD FLTBINDFLOWF:FLOWFILTERNAME="ff_any_ims_paas", FILTERNAME="f_any";
ADD FLTBINDFLOWF:FLOWFILTERNAME="ff_any_ims_paas", FILTERNAME="f_any_v6";
ADD PROTBINDFLOWF:FLOWFILTERNAME="ff_any_ims_drop", PROTOCOLNAME="sip_esp_encrypted";
```

//配置动作属性为通过。非加密的标准IMS业务报文正常转发。

```
ADD PCCACTIONPROP:PCCACTPROPNAME="pcc_pass", UPINITUPGATE=PASS, UPINITDNGATE=PASS, DNINITUPGATE=PASS, DNINITDNGATE=PASS;
```

//配置动作属性为阻断，对加密业务报文进行阻断。

```
ADD PCCACTIONPROP:PCCACTPROPNAME="pcc_drop", UPINITUPGATE=DISCARD, UPINITDNGATE=DISCARD, DNINITUPGATE=DISCARD, DNINITDNGATE=DISCARD;
```

//配置PCC策略组。当需要计费时，请配置参数 “URR组名称（URRGROUPNAME）” 。

```
ADD PCCPOLICYGRP:PCCPOLICYGRPNM="pccg_ims_drop", 
URRGROUPNAME="urrgp_test"
, PCCACTPROPNAME="pcc_drop", FUPSESSIONEXC=DISABLE, EVENTCHARGEFLAG=DISABLE, EVENTCHGPOINT=RESPONSE, MMSCHARGETYPE=SINGLE, SIGNALASSOCIATE=ENABLE, ADCMUTEFLAG=ENABLE, HTTP2DEGRADESW=INHERIT, TOKENFUNCFLAG=DISABLE;
ADD PCCPOLICYGRP:PCCPOLICYGRPNM="pccg_ims_paas", 
URRGROUPNAME="urrgp_test"
, PCCACTPROPNAME="pcc_pass", FUPSESSIONEXC=DISABLE, EVENTCHARGEFLAG=DISABLE, EVENTCHGPOINT=RESPONSE, MMSCHARGETYPE=SINGLE, SIGNALASSOCIATE=ENABLE, ADCMUTEFLAG=ENABLE, HTTP2DEGRADESW=INHERIT, TOKENFUNCFLAG=DISABLE;
```

//配置业务的使用规则。

```
ADD RULE:RULENAME="rule_ims_drop", POLICYTYPE=PCC, FLOWFILTERNAME="ff_any_ims_drop", PRIORITY=10, POLICYNAME="pccg_ims_drop", FILTERINGMODE=FLOWFILTER, RULESPECTYPE=DEFAULT;
ADD RULE:RULENAME="rule_ims_paas", POLICYTYPE=PCC, FLOWFILTERNAME="ff_any_ims_paas", PRIORITY=4294967295, POLICYNAME="pccg_ims_paas", FILTERINGMODE=FLOWFILTER, RULESPECTYPE=DEFAULT;
```

//配置业务策略组合。

```
ADD USERPROFILE:USERPROFILENAME="up_proxy_sgw_bit1559", ALIASMARKFLAG=DISABLE, DDOSCHECK=DISABLE, CAPMODETHRES=0, LOCRPTSWITCH=INHERIT, TETHERINGMAXNUM=255, V6TCPMSSVALUE=0, V4TCPMSSVALUE=0, UPSPECTYPE=DEFAULT;
SET URRGRPBINDING:USERPROFILENAME="up_proxy_sgw_bit1559", TCPCHGPROFLAG=NOTCONFIG;
ADD RULEBINDING:USERPROFILENAME="up_proxy_sgw_bit1559", RULENAME="rule_ims_drop", POLICYTYPE=PCC;
ADD RULEBINDING:USERPROFILENAME="up_proxy_sgw_bit1559", RULENAME="rule_ims_paas", POLICYTYPE=PCC;
SET REFRESHSRV:REFRESHTYPE=USERPROFILE;
```
