## [操作场景](#ZH-CN_OPI_0172467890)

- 当网络中有PCF，用户激活为PCC用户，需要为用户或用户业务提供灵活策略控制时，需要开启动态PCC功能，并根据业务情况来部署预定义规则/预定义规则组或动态规则。
    - 预定义规则/预定义规则组：PCF上只配置预定义规则/预定义规则组名称，相应的QoS管理、计费控制策略等在UNC上通过rule或者UserProfile配置。
    - 动态规则：在PCF上配置相应的QoS管理、计费控制策略等，UNC上只需开启PCC功能开关。

- 当通过特定DNN激活的用户需要激活为PCC用户，而网络中未部署PCF时或不需要UNC与PCF交互时，需要开启本地PCC功能，并在本地配置静态规则，策略匹配时按照rule的全局优先级进行匹配。
    - UNC上开启本地PCC功能时，需要在UNC上配置rule和UserProfile，本处在UNC上仅开启PCC功能开关，不关注具体规则内容配置，具体规则内容需要根据具体业务策略确定。

> **说明**
> 适用于AMF、SMF。UNC作为GGSN-C/PGW-C时，支持通过N7接口与PCF交互获取策略信息。

任务一：UNC上规划AMF不开启PCC功能，SMF全局和指定DNN使能动态PCC功能，且当用户使用DNN为huawei.com激活时触发PCC功能，用户激活为动态PCC用户，且使用本地配置的规则按照全局优先级进行策略匹配。

任务二：UNC上规划AMF开启PCC功能，SMF全局使能动态PCC功能，DNN“apn-test1”下使能动态PCC功能，DNN“apn-test2”下继承全局使能开关，DNN“apn-another”开启本地PCC功能。PCF采用主备容灾组网。

任务三：UNC上规划AMF开启PCC功能，SMF全局使能PCC功能，DNN“apn-test1”下使能PCC功能，且基于DNN和SERVINGSCOPE选择PCF。DNN“apn-test2”下继承全局使能开关，DNN“apn-another”关闭PCC功能。PCF采用主备容灾组网。

脚本

//任务一：UNC上规划AMF不开启PCC功能，SMF全局和指定DNN使能动态PCC功能，且当用户使用DNN为huawei.com激活时触发PCC功能，用户激活为动态PCC用户，且使用本地配置的规则按照全局优先级进行策略匹配。

//打开License配置开关。

```
SET LICENSESWITCH: LICITEM="LKV2PCCBF01", SWITCH=ENABLE;
SET LICENSESWITCH: LICITEM="LKV3W9SPCC11", SWITCH=ENABLE;
```

//配置SMF的全局缺省的PCC开关。

```
SET PCCFUNC:HOMEPCCSWITCH=ENABLE,ROAMPCCSWITCH=ENABLE,VISITPCCSWITCH=ENABLE,COMPOLICY=LOCALAAA;
```

//配置指定DNN的PCC开关。如下脚本以支持根据PCF发送的UpdateNotify/TerminateNotify携带的ResourceURI触发PCF重选为例（NTFRSRURI=ENABLE）。如下脚本以支持PCF根据GROUPID为例（PCFLBPARA=GROUPID）。

```
ADD PCCTEMPLATE:PCCTEMPNAME="pcctemplate",SELPEERFAILACT=DEFAULT,INITIALFAILACT=FORBIDDEN,UPDATEFAILACT=INHERIT,NTFRSRURI=ENABLE,N7FAILOVERSW=ENABLE,PCFLBPARA=GROUPID;
SET APNPCCFUNC:APN="huawei.com",HOMEPCCSWITCH=ENABLE,ROAMPCCSWITCH=ENABLE,VISITPCCSWITCH=ENABLE,PCCTEMPLATE="pcctemplate";
```

//配置PCC业务规则。如配置的rule仅下发给辅锚点UPF，则需要将“RULERANGE”设置为“LOCAL”；仅下发给主锚点UPF则设置为“CENTRAL”。

```
ADD PCCPOLICYGRP:PCCPOLICYGRPNM="pccpolicy_group",URRGROUPNAME="urrgrp_01",FUPSESSIONEXC=ENABLE;
ADD RULE:RULENAME="rule_test",POLICYTYPE=PCC,POLICYNAME="pccpolicy_group",RULERANGE=ALL;
```

//配置业务策略组合。如配置的Userprofile仅下发给辅锚点UPF，则需要将“PROFILERANGE”设置为“LOCAL”；仅下发给主锚点UPF则设置为“CENTRAL”。

```
ADD USERPROFILE:USERPROFILENAME="up_test",UPTYPE=PCC,PROFILERANGE=ALL;
ADD RULEBINDING:USERPROFILENAME="up_test",RULENAME="rule_test";
ADD USRPROFGROUP:USERPROFGNAME="upg_test";
ADD UPBINDUPG:USERPROFGNAME="upg_test",UPBINDTYPE=DEFAULT,USERPROFILENAME="up_test";
ADD APNUSRPROFG:APN="huawei.com",USERPROFGNAME="upg_test";
```

//任务二：UNC上规划AMF开启PCC功能，SMF全局使能PCC功能，DNN实例“apn-test1”下使能PCC功能，DNN实例“apn-test2”下继承全局使能开关，DNN实例“apn-another”关闭PCC功能。

//打开License配置开关。

```
SET LICENSESWITCH: LICITEM="LKV2PCCBF01", SWITCH=ENABLE;
SET LICENSESWITCH: LICITEM="LKV3W9SPCC11", SWITCH=ENABLE;
```

//配置AMF选择PCF的策略信息。

```
ADD PCFSELPLCY: SUBRANGE=HOME_USER, TGTPLMNSW=NO, SUPISW=YES, GPSISW=NO, NSSW=NO,  RETRYSW=YES, DESC="for home user";
```

//在AMF上配置应用网络侧规划的AM策略和UE策略的用户群。

```
ADD AMUEPLCYCTRL: SUBRANGE=HOME_USER, NSGRPID=0, ISAMASSOC=YES, ISUEASSOC=YES, DESC="Enable UE Policy for Home Subscribers";
```

//在AMF上配置本地的RFSP索引，并且本地配置的RFSP索引优先级高于签约的RFSP索引。

```
ADD NGMMSUBDATA: SUBRANGE=HOME_USER, RATRESTRICT=EUTRA-0&NR-1, CORERESTRICT=FIFTHGC-1, RFSPPRI=CFG_FIRST, RFSPINDEX=5;
```

//配置SMF的全局粒度的PCC开关。

```
SET PCCFUNC:HOMEPCCSWITCH=ENABLE,ROAMPCCSWITCH=ENABLE,VISITPCCSWITCH=ENABLE,COMPOLICY=LOCALAAA;
```

//配置三个DNN下的PCC开关，其中DNN实例apn-test1的PCC开关为“ENABLE”，DNN实例apn-test2的PCC开关为“INHERIT”，关闭DNN实例apn-another的PCC开关。

```
SET APNPCCFUNC:APN="apn-test1",HOMEPCCSWITCH=ENABLE,ROAMPCCSWITCH=ENABLE,VISITPCCSWITCH=ENABLE;
SET APNPCCFUNC:APN="apn-test2",HOMEPCCSWITCH=INHERIT,ROAMPCCSWITCH=INHERIT,VISITPCCSWITCH=INHERIT;
SET APNPCCFUNC:APN="apn-another",HOMEPCCSWITCH=DISABLE,ROAMPCCSWITCH=DISABLE,VISITPCCSWITCH=DISABLE;
```

//配置AMF支持使用PCF通过Location信元携带的地址进行通信。

```
SET AMFPEERSELFUNC: NFLOCATION=PCF-1;
```

//任务三：UNC上规划AMF开启PCC功能，SMF全局使能动态PCC功能，DNN“apn-test1”下使能动态PCC功能，且基于DNN和SERVINGSCOPE选择PCF。DNN“apn-test2”下继承全局使能开关，DNN“apn-another”关闭PCC功能（开启本地PCC功能）。PCF采用主备容灾组网。

//打开License配置开关。

```
SET LICENSESWITCH: LICITEM="LKV2PCCBF01", SWITCH=ENABLE;
SET LICENSESWITCH: LICITEM="LKV3W9SPCC11", SWITCH=ENABLE;
```

//配置AMF选择PCF的策略信息。

```
ADD PCFSELPLCY: SUBRANGE=HOME_USER, TGTPLMNSW=NO, SUPISW=YES, GPSISW=NO, NSSW=NO,  RETRYSW=YES, DESC="for home user";
```

//在AMF上配置应用网络侧规划的AM策略和UE策略的用户群。

```
ADD AMUEPLCYCTRL: SUBRANGE=HOME_USER, NSGRPID=0, ISAMASSOC=YES, ISUEASSOC=YES, DESC="Enable UE Policy for Home Subscribers";
```

//在AMF上配置本地的RFSP索引，并且本地配置的RFSP索引优先级高于签约的RFSP索引。

```
ADD NGMMSUBDATA: SUBRANGE=HOME_USER, RATRESTRICT=EUTRA-0&NR-1, CORERESTRICT=FIFTHGC-1, RFSPPRI=CFG_FIRST, RFSPINDEX=5;
```

//配置SMF的全局缺省的PCC开关。

```
SET PCCFUNC:HOMEPCCSWITCH=ENABLE,ROAMPCCSWITCH=ENABLE,VISITPCCSWITCH=ENABLE,COMPOLICY=LOCALAAA;
```

//配置指定DNN“apn-test1”的PCC开关。如下脚本以支持根据PCF发送的UpdateNotify/TerminateNotify携带的ResourceURI触发PCF重选为例（NTFRSRURI=ENABLE）。如下脚本以支持PCF根据PRIORITY为例（PCFLBPARA=PRIORITY）。

```
ADD PCCTEMPLATE:PCCTEMPNAME="pcctemplate",SELPEERFAILACT=DEFAULT,INITIALFAILACT=FORBIDDEN,UPDATEFAILACT=INHERIT,NTFRSRURI=ENABLE,N7FAILOVERSW=ENABLE,PCFLBPARA=PRIORITY;
SET APNPCCFUNC:APN="apn-test1",HOMEPCCSWITCH=ENABLE,ROAMPCCSWITCH=ENABLE,VISITPCCSWITCH=ENABLE,PCCTEMPLATE="pcctemplate";
```

//配置三个DNN下的PCC开关，其中DNN实例apn-test1的PCC开关为“ENABLE”，DNN实例apn-test2的PCC开关为“INHERIT”，关闭DNN实例apn-another的PCC开关。

```
SET APNPCCFUNC:APN="apn-test1",HOMEPCCSWITCH=ENABLE,ROAMPCCSWITCH=ENABLE,VISITPCCSWITCH=ENABLE,PCFSELECTMODE=DNN-1&SERVINGSCOPE-1;
SET APNPCCFUNC:APN="apn-test2",HOMEPCCSWITCH=INHERIT,ROAMPCCSWITCH=INHERIT,VISITPCCSWITCH=INHERIT;
SET APNPCCFUNC:APN="apn-another",HOMEPCCSWITCH=DISABLE,ROAMPCCSWITCH=DISABLE,VISITPCCSWITCH=DISABLE;
```

//配置AMF支持使用PCF通过Location信元携带的地址进行通信。

```
SET AMFPEERSELFUNC: NFLOCATION=PCF-1;
```
